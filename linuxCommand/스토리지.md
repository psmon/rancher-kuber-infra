
# 초기화

fdisk -l

mkfs.ext4 /dev/sda1 - 포맺

mkdir /media/usb

# 마운트

## ntfs or vfat

mount -t ext4 /dev/sda1 /media/usb

mount -t vfat /dev/sda1 /media/usb

mount -t ntfs /dev/sda1 /media/usb

umount /dev/sda1

# 자동마운트
cp /etc/fstab /etc/fstab.bak

vi /etc/fstab

/dev/sda1 /media/usb ext4 defaults 0 0
